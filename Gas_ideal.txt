#include <Wire.h>
#include <Adafruit_BMP280.h>

Adafruit_BMP280 bmp; 


unsigned long ultimoMinuto = 0;
int minutoAtual = 0;

void setup() {

  
  Serial.begin(9600);

  Serial.println(F("BMP280 teste"));

  
  if (!bmp.begin(0x76)) {

    Serial.println(F("Nao foi possivel encontrar um sensor BMP280 valido!"));
    
    while (1) delay(10);
  }
}

void loop() {

  
  if (millis() - ultimoMinuto >= 60000 || minutoAtual == 0) {

    ultimoMinuto = millis();

    Serial.println();
    Serial.println("==============================");
    Serial.print("Tempo: ");
    Serial.print(minutoAtual);
    Serial.println(" minuto(s)");
    Serial.println("==============================");

    
    for (int i = 1; i <= 10; i++) {

      float temperatura = bmp.readTemperature();


      Serial.print(temperatura);
      Serial.print(",");


      
      delay(50);
    }
    for (int j = 1; j <= 10; j++) {

      

      
      float pressao = bmp.readPressure() / 100.0;

     
     


      
      Serial.print(pressao);
      Serial.print(",");
    

      delay(50);
    }

    minutoAtual++;
  }
}
